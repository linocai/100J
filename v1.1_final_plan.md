### 4.2 Apple Sign in 端点

新建 `app/services/apple_auth_service.py`：

```python
import json
import time
import urllib.request
from functools import lru_cache
from typing import Optional

import jwt
from sqlalchemy import select
from sqlalchemy.orm import Session

from app.core.config import get_settings
from app.core.errors import AppError
from app.models import User
from app.services.auth_service import create_default_spaces, issue_tokens

APPLE_JWKS_URL = "https://appleid.apple.com/auth/keys"
APPLE_ISS = "https://appleid.apple.com"

@lru_cache(maxsize=1)
def _jwks() -> dict:
    # 真正生产实现要带 24h TTL 缓存 + ETag
    with urllib.request.urlopen(APPLE_JWKS_URL, timeout=5) as r:
        return json.load(r)

def _verify_apple_id_token(id_token: str, expected_audience: str) -> dict:
    headers = jwt.get_unverified_header(id_token)
    kid = headers["kid"]
    jwks = _jwks()
    key = next((k for k in jwks["keys"] if k["kid"] == kid), None)
    if not key:
        raise AppError(401, "unauthorized", "Apple key not found")
    public_key = jwt.algorithms.RSAAlgorithm.from_jwk(json.dumps(key))
    try:
        return jwt.decode(
            id_token, public_key,
            algorithms=["RS256"],
            audience=expected_audience,
            issuer=APPLE_ISS,
        )
    except jwt.InvalidTokenError as exc:
        raise AppError(401, "unauthorized", f"Invalid Apple identity token: {exc}")

def sign_in_with_apple(
    db: Session,
    id_token: str,
    email_hint: Optional[str],
    full_name_hint: Optional[str],
    bundle_id: str,
) -> User:
    settings = get_settings()
    if bundle_id not in settings.apple_allowed_audiences:
        raise AppError(401, "unauthorized", "Apple audience not allowed")
    claims = _verify_apple_id_token(id_token, expected_audience=bundle_id)
    sub = claims["sub"]
    email = (claims.get("email") or email_hint or "").lower() or None

    user = db.scalar(select(User).where(User.apple_user_id == sub, User.deleted_at.is_(None)))
    if user is None and email:
        user = db.scalar(select(User).where(User.email == email, User.deleted_at.is_(None)))
        if user:
            user.apple_user_id = sub

    if user is None:
        user = User(
            apple_user_id=sub,
            email=email,
            display_name=full_name_hint or email or "100J User",
            timezone="Asia/Shanghai",
            password_hash=None,
        )
        db.add(user)
        db.flush()
        create_default_spaces(db, user)
    db.commit()
    db.refresh(user)
    return user
```

### 4.3 Email OTP（无密码登录）

新建 `app/services/email_otp_service.py`：

```python
import secrets
import string
from datetime import datetime, timedelta, timezone
from hashlib import sha256

from sqlalchemy import select
from sqlalchemy.orm import Session

from app.core.errors import AppError
from app.models.email_otp import EmailOTPCode
from app.models import User
from app.services.auth_service import create_default_spaces

OTP_LENGTH = 6
OTP_TTL_MINUTES = 10
OTP_MAX_ATTEMPTS = 5

def _hash(code: str) -> str:
    return sha256(code.encode("utf-8")).hexdigest()

def request_code(db: Session, email: str, send: callable) -> None:
    code = "".join(secrets.choice(string.digits) for _ in range(OTP_LENGTH))
    db.add(EmailOTPCode(
        email=email.lower(),
        code_hash=_hash(code),
        expires_at=datetime.now(timezone.utc) + timedelta(minutes=OTP_TTL_MINUTES),
    ))
    db.commit()
    # send 是注入的发件实现（dev = print，prod = SES/Resend）
    send(email=email, code=code)

def verify_code(db: Session, email: str, code: str) -> User:
    row = db.scalar(
        select(EmailOTPCode)
        .where(
            EmailOTPCode.email == email.lower(),
            EmailOTPCode.consumed_at.is_(None),
            EmailOTPCode.expires_at > datetime.now(timezone.utc),
        )
        .order_by(EmailOTPCode.created_at.desc())
    )
    if not row:
        raise AppError(401, "unauthorized", "Code expired or not requested.")
    if row.attempts >= OTP_MAX_ATTEMPTS:
        raise AppError(429, "rate_limited", "Too many attempts.")
    row.attempts += 1
    if row.code_hash != _hash(code):
        db.commit()
        raise AppError(401, "unauthorized", "Invalid code.")
    row.consumed_at = datetime.now(timezone.utc)

    user = db.scalar(select(User).where(User.email == email.lower(), User.deleted_at.is_(None)))
    if not user:
        user = User(email=email.lower(), display_name=email.split("@")[0],
                    password_hash=None, timezone="Asia/Shanghai")
        db.add(user); db.flush()
        create_default_spaces(db, user)
    db.commit()
    db.refresh(user)
    return user
```

### 8.2 离线 mutation 队列

```swift
public actor MutationQueue {
    struct PendingMutation: Codable, Identifiable {
        enum Kind: String, Codable { case createTask, updateTask, completeTask, createNote, createCalendar, … }
        let id: UUID; let kind: Kind; let payload: Data; let createdAt: Date
    }
    private var pending: [PendingMutation] = []
    private let storage: URL    // ~/Library/Application Support/100J/queue.json

    func enqueue(_ m: PendingMutation) async { pending.append(m); await persist() }

    func replay(api: APIClient) async {
        while let next = pending.first {
            do {
                try await execute(next, api: api)
                pending.removeFirst(); await persist()
            } catch APIError.network {
                return  // 等下一次 reachable
            } catch let APIError.http(status, _) where (400..<500).contains(status) {
                // 永久失败：丢弃但记 audit log
                pending.removeFirst(); await persist()
            }
        }
    }
}
```

挂载：`APIClient` 收到 `URLError.notConnectedToInternet` 时把 mutation 推入 queue，UI 立刻乐观更新；`NWPathMonitor` 监听到联网回来调 `MutationQueue.replay`。

## 10. 风险与缓解

| 风险 | 影响 | 缓解 |
|---|---|---|
| Sign in with Apple 在自部署域名上需要 Service ID + Return URLs 配置，单设备登录路径不需要 | 申请 1-2 天 | 用 Native Apple ID 流程（不需要 web service ID），后端只验签 id_token 即可，无需 web return URL |
| Apple JWKS 拉取被墙 | 登录失败 | 后端起一个 cron daily 拉 JWKS 缓存到 redis/disk，端点优先读缓存 |
| Liquid Glass 仅 macOS/iOS 26 可用 | 老系统观感降级 | 抽 `GlassBackground` modifier，26 用 `.glassEffect`，老系统 fallback `.regularMaterial` |
| iOS 26 widget API 变更 | widget 编译失败 | 用 `if #available(iOS 17.0, *)` 多分支；目标 deployment iOS 17，新特性 opt-in |
| 邮件 OTP 钓鱼风险 | 用户邮箱被恶意请求 | `request_email_otp` 加 captcha（生产）或固定 5/min/IP；模板里写明"如果不是你本人请忽略" |
| 离线队列冲突（同一资源被改两次） | 后端拒一笔，UI 看起来"操作丢了" | 重放失败 → 提示用户"以下操作未同步"，提供"重试 / 丢弃"二选一，写入 audit log |
| Menu Bar Extra 状态点判断错（误判离线） | 用户误以为坏了 | sync status 用 `NWPathMonitor` + 上次 successful API call timestamp 双因子，>120s 无成功才标 stale |
